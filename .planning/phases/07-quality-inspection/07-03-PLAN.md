```java
@Component
public class ScriptCompletenessRule {
    
    private static final double COMPLETION_THRESHOLD = 0.8; // 80% 阈值
    
    @Autowired
    private DialogTreeManager treeManager;
    
    /**
     * 检查话术完整性
     * @param sessionId 会话 ID
     * @return 话术完成结果
     */
    public ScriptCompletionResult check(String sessionId) {
        DialogState state = treeManager.getCurrentState(sessionId);
        DialogTree tree = treeManager.getDialogTree(state.getDialogTreeId());
        
        // 获取必需节点
        List<String> requiredNodes = tree.getRequiredNodes();
        
        // 获取已完成节点（从历史中筛选必需节点）
        List<String> completedNodes = state.getHistory()
            .stream()
            .filter(nodeId -> requiredNodes.contains(nodeId))
            .collect(Collectors.toList());
        
        // 计算完成率
        int completed = completedNodes.size();
        int total = requiredNodes.size();
        double rate = total > 0 ? (double) completed / total : 0.0;
        
        // 计算缺失节点
        List<String> missingNodes = requiredNodes.stream()
            .filter(nodeId -> !completedNodes.contains(nodeId))
            .collect(Collectors.toList());
        
        return ScriptCompletionResult.builder()
            .completionRate(rate)
            .completedCount(completed)
            .totalCount(total)
            .passed(rate >= COMPLETION_THRESHOLD)
            .missingNodes(missingNodes)
            .build();
    }
}
```

```java
@Component
public class ViolationKeywordsRule {
    
    @Autowired
    private SensitiveWordService sensitiveWordService;
    
    /**
     * 检测对话历史中的违规词
     * @param sessionId 会话 ID
     * @return 违规记录列表
     */
    public List<ViolationRecord> detect(String sessionId) {
        DialogState state = treeManager.getCurrentState(sessionId);
        
        // 构建对话文本（从历史记录）
        String transcript = buildTranscript(state);
        
        // 检测违规词
        List<String> violations = sensitiveWordService.detectViolations(transcript);
        
        // 转换为违规记录
        return violations.stream()
            .map(word -> ViolationRecord.builder()
                .word(word)
                .severity(classifySeverity(word))
                .timestamp(LocalDateTime.now())
                .context(extractContext(transcript, word))
                .build())
            .collect(Collectors.toList());
    }
    
    /**
     * 判断违规词严重程度
     * - 严重：保本、保收益、刚性兑付、无风险
     * - 一般：其他违规词
     */
    private ViolationSeverity classifySeverity(String word) {
        List<String> severeWords = Arrays.asList("保本", "保收益", "刚性兑付", "无风险");
        return severeWords.contains(word) ? ViolationSeverity.SEVERE : ViolationSeverity.GENERAL;
    }
    
    /**
     * 提取上下文（违规词前后 20 个字）
     */
    private String extractContext(String transcript, String word) {
        int index = transcript.indexOf(word);
        if (index == -1) return "";
        
        int start = Math.max(0, index - 20);
        int end = Math.min(transcript.length(), index + word.length() + 20);
        return transcript.substring(start, end);
    }
    
    private String buildTranscript(DialogState state) {
        // TODO: 从 DialogState 构建对话文本
        // MVP 阶段可以先返回空字符串或简单拼接
        return state.getHistory().stream().collect(Collectors.joining(" "));
    }
}
```

```java
@Component
public class QualityScoringRule {
    
    private static final int SCRIPT_COMPLETION_DEDUCTION = 20;  // 话术不完整扣 20 分
    private static final int GENERAL_VIOLATION_DEDUCTION = 5;   // 一般违规扣 5 分
    private static final int SEVERE_VIOLATION_DEDUCTION = 20;   // 严重违规扣 20 分
    private static final int PASS_THRESHOLD = 70;               // 通过阈值 70 分
    
    /**
     * 计算质检分数
     * @param scriptResult 话术完成结果
     * @param violations 违规记录列表
     * @return 质检报告
     */
    public QualityReport calculate(ScriptCompletionResult scriptResult,
                                   List<ViolationRecord> violations) {
        int score = 100;  // 起点 100 分
        List<String> reasons = new ArrayList<>();
        
        // 话术完整性扣分
        if (!scriptResult.getPassed()) {
            score -= SCRIPT_COMPLETION_DEDUCTION;
            reasons.add(String.format("话术完整性不足 (%.1f%%)", 
                scriptResult.getCompletionRate() * 100));
        }
        
        // 违规词扣分（去重）
        Set<String> uniqueViolations = violations.stream()
            .map(ViolationRecord::getWord)
            .collect(Collectors.toSet());
        
        for (String word : uniqueViolations) {
            ViolationRecord record = violations.stream()
                .filter(v -> v.getWord().equals(word))
                .findFirst()
                .orElseThrow();
            
            if (record.getSeverity() == ViolationSeverity.SEVERE) {
                score -= SEVERE_VIOLATION_DEDUCTION;
                reasons.add("严重违规词：" + word);
            } else {
                score -= GENERAL_VIOLATION_DEDUCTION;
                reasons.add("一般违规词：" + word);
            }
        }
        
        // 确保分数不低于 0
        score = Math.max(0, score);
        
        // 判断是否通过（分数≥70 且无严重违规）
        boolean hasSevereViolation = violations.stream()
            .anyMatch(v -> v.getSeverity() == ViolationSeverity.SEVERE);
        
        boolean passed = score >= PASS_THRESHOLD && !hasSevereViolation;
        
        return QualityReport.builder()
            .score(score)
            .passed(passed)
            .reason(String.join("; ", reasons))
            .violations(violations)
            .inspectionTime(LocalDateTime.now())
            .build();
    }
}
```
