```java
@Configuration
public class QualityInspectionConfig {
    
    /**
     * 质检异步执行线程池
     * 
     * 配置说明：
     * - CorePoolSize: 4 (基础并发)
     * - MaxPoolSize: 10 (最大并发)
     * - QueueCapacity: 50 (排队容量)
     * - KeepAliveSeconds: 60 (空闲存活)
     * - ThreadNamePrefix: "quality-inspection-" (线程名前缀)
     */
    @Bean(name = "qualityInspectionExecutor")
    public ThreadPoolTaskExecutor qualityExecutor() {
        ThreadPoolTaskExecutor executor = new ThreadPoolTaskExecutor();
        executor.setCorePoolSize(4);
        executor.setMaxPoolSize(10);
        executor.setQueueCapacity(50);
        executor.setThreadNamePrefix("quality-inspection-");
        executor.setKeepAliveSeconds(60);
        executor.setAwaitTerminationSeconds(60);
        executor.setRejectedExecutionHandler(new ThreadPoolExecutor.CallerRunsPolicy());
        executor.initialize();
        return executor;
    }
}
```

配置理由：
- CorePoolSize=4：满足基础并发需求（100+ 并发录制会话）
- MaxPoolSize=10：支持突发流量
- QueueCapacity=50：防止内存溢出
- CallerRunsPolicy：队列满时由调用线程执行（背压机制）

```java
@Component
public class QualityInspectionAgent {
    
    private static final Logger logger = LoggerFactory.getLogger(QualityInspectionAgent.class);
    
    @Autowired
    private DialogTreeManager treeManager;
    
    @Autowired
    private ScriptCompletenessRule completenessRule;
    
    @Autowired
    private ViolationKeywordsRule violationRule;
    
    @Autowired
    private QualityScoringRule scoringRule;
    
    /**
     * 执行质检
     * @param sessionId 会话 ID
     * @return 质检报告
     */
    public QualityReport inspect(String sessionId) {
        long startTime = System.currentTimeMillis();
        logger.info("开始质检：sessionId={}", sessionId);
        
        try {
            // 1. 获取对话状态和树
            DialogState state = treeManager.getCurrentState(sessionId);
            if (state == null) {
                return QualityReport.timeout("会话状态不存在：" + sessionId);
            }
            
            // 2. 执行话术完整性检测
            ScriptCompletionResult scriptResult = completenessRule.check(sessionId);
            logger.debug("话术完整性检测完成：sessionId={}, rate={}", 
                sessionId, scriptResult.getCompletionRate());
            
            // 3. 执行违规词检测
            List<ViolationRecord> violations = violationRule.detect(sessionId);
            logger.debug("违规词检测完成：sessionId={}, count={}", 
                sessionId, violations.size());
            
            // 4. 计算评分
            QualityReport report = scoringRule.calculate(scriptResult, violations);
            
            // 5. 设置质检耗时
            long duration = System.currentTimeMillis() - startTime;
            report.setDurationMs(duration);
            
            logger.info("质检完成：sessionId={}, score={}, passed={}, duration={}ms", 
                sessionId, report.getScore(), report.getPassed(), duration);
            
            return report;
            
        } catch (Exception e) {
            logger.error("质检异常：sessionId={}", sessionId, e);
            return QualityReport.timeout("质检异常：" + e.getMessage());
        }
    }
}
```

```java
@Component
public class QualityInspectionTrigger {
    
    private static final Logger logger = LoggerFactory.getLogger(QualityInspectionTrigger.class);
    
    @Autowired
    private QualityInspectionAgent agent;
    
    @Qualifier("qualityInspectionExecutor")
    @Autowired
    private ThreadPoolTaskExecutor qualityExecutor;
    
    /**
     * 触发质检（异步执行）
     * @param sessionId 会话 ID
     * @return CompletableFuture<QualityReport>
     */
    public CompletableFuture<QualityReport> triggerInspection(String sessionId) {
        logger.info("触发质检：sessionId={}", sessionId);
        
        CompletableFuture<QualityReport> future = CompletableFuture.supplyAsync(() -> {
            return agent.inspect(sessionId);
        }, qualityExecutor);
        
        // 设置 3 分钟超时
        return future.orTimeout(180, TimeUnit.SECONDS)
            .exceptionally(ex -> {
                logger.error("质检超时或失败：sessionId={}", sessionId, ex);
                return QualityReport.timeout("质检超时或失败：" + ex.getMessage());
            });
    }
    
    /**
     * 同步触发质检（阻塞等待结果）
     * @param sessionId 会话 ID
     * @return 质检报告
     */
    public QualityReport triggerInspectionSync(String sessionId) {
        try {
            return triggerInspection(sessionId).get(180, TimeUnit.SECONDS);
        } catch (TimeoutException e) {
            logger.error("质检超时：sessionId={}", sessionId, e);
            return QualityReport.timeout("质检超时");
        } catch (InterruptedException | ExecutionException e) {
            logger.error("质检异常：sessionId={}", sessionId, e);
            return QualityReport.timeout("质检异常：" + e.getMessage());
        }
    }
}
```
