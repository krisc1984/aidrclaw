```typescript
// InviteGenerator.tsx
import QRCode from 'qrcode.react';
import { Button } from '@/components/ui/button';
import { Input } from '@/components/ui/input';
import { Card, CardContent, CardHeader, CardTitle } from '@/components/ui/card';

interface InviteGeneratorProps {
  sessionId: string;
  token: string;
 理财 ManagerName: string;
  productName: string;
}

export const InviteGenerator = ({ sessionId, token, 理财 ManagerName, productName }: InviteGeneratorProps) => {
  const inviteUrl = `${window.location.origin}/record/${sessionId}?token=${token}`;
  
  const copyLink = async () => {
    await navigator.clipboard.writeText(inviteUrl);
    alert('链接已复制到剪贴板');
  };
  
  return (
    <Card>
      <CardHeader>
        <CardTitle>双录邀请</CardTitle>
      </CardHeader>
      <CardContent className="space-y-4">
        <div className="text-sm text-gray-600">
          <div>理财经理：{理财 ManagerName}</div>
          <div>产品：{productName}</div>
        </div>
        
        {/* 二维码 */}
        <div className="flex justify-center">
          <QRCode value={inviteUrl} size={256} level="H" />
        </div>
        
        {/* 邀请链接 */}
        <div className="space-y-2">
          <div className="text-sm font-medium">邀请链接：</div>
          <div className="flex gap-2">
            <Input value={inviteUrl} readOnly className="flex-1" />
            <Button onClick={copyLink}>复制</Button>
          </div>
        </div>
        
        <div className="text-xs text-gray-500">
          提示：客户可扫描二维码或点击链接进入双录界面
        </div>
      </CardContent>
    </Card>
  );
};
```

创建客户端双录界面（集成虚拟坐席）：

```typescript
// ClientRecord.tsx
import { useEffect, useState } from 'react';
import { useParams, useSearchParams } from 'react-router-dom';
import axios from 'axios';
import { VirtualAgentInterface } from '@/components/virtual-agent/VirtualAgentInterface';
import { Alert, AlertDescription, AlertTitle } from '@/components/ui/alert';
import { AlertCircle } from 'lucide-react';

export const ClientRecord = () => {
  const { sessionId } = useParams<{ sessionId: string }>();
  const [searchParams] = useSearchParams();
  const token = searchParams.get('token');
  
  const [session, setSession] = useState<any>(null);
  const [error, setError] = useState<string | null>(null);
  const [warnings, setWarnings] = useState<string[]>([]);
  
  useEffect(() => {
    // 验证会话 Token
    axios.get(`/api/session/${sessionId}`, {
      headers: { Authorization: `Bearer ${token}` }
    })
    .then(res => setSession(res.data))
    .catch(err => setError('无效的双录邀请'));
  }, [sessionId, token]);
  
  // 监听质检警告（实时提示）
  useEffect(() => {
    const ws = new WebSocket(`ws://localhost:8080/ws/quality/${sessionId}`);
    ws.onmessage = (event) => {
      const data = JSON.parse(event.data);
      if (data.warnings) {
        setWarnings(data.warnings);
      }
    };
    return () => ws.close();
  }, [sessionId]);
  
  if (error) {
    return (
      <div className="flex items-center justify-center min-h-screen">
        <Alert variant="destructive" className="max-w-md">
          <AlertCircle className="h-4 w-4" />
          <AlertTitle>错误</AlertTitle>
          <AlertDescription>{error}</AlertDescription>
        </Alert>
      </div>
    );
  }
  
  return (
    <div className="min-h-screen bg-gray-50">
      {/* 顶部信息 */}
      <header className="bg-white border-b p-4">
        <div className="max-w-4xl mx-auto">
          <h1 className="text-xl font-bold">双录系统</h1>
          <p className="text-sm text-gray-600">
            理财经理：{session?.理财 ManagerName} | 产品：{session?.productName}
          </p>
        </div>
      </header>
      
      {/* 主内容区 */}
      <main className="max-w-4xl mx-auto p-6">
        {/* 实时警告 */}
        {warnings.length > 0 && (
          <Alert className="mb-4">
            <AlertCircle className="h-4 w-4" />
            <AlertTitle>温馨提示</AlertTitle>
            <AlertDescription>
              <ul className="list-disc list-inside">
                {warnings.map((warning, index) => (
                  <li key={index}>{warning}</li>
                ))}
              </ul>
            </AlertDescription>
          </Alert>
        )}
        
        {/* 虚拟坐席界面 */}
        <VirtualAgentInterface 
          sessionId={sessionId!}
          onWarning={(warning) => setWarnings([...warnings, warning])}
        />
      </main>
    </div>
  );
};
```
